# Get the available leagues
available_leagues = sd.FotMob.available_leagues()
exclude_leagues = ["ARG-Liga Profesional", "BRA-Brasileirão", "INT-Women's World Cup", "INT-World Cup", 'INT-European Championship',
                   "RUS-Premier League", "USA-Major League Soccer", "POR-Liga Portugal"]
available_leagues = [league for league in available_leagues if league not in exclude_leagues]

# Initialize an empty list to store the schedules
all_schedules = []

# Loop through each available league
for league in available_leagues:
    try:
        # Special case for ENG-Community Shield
        season = '2021/2022'
        if league in ['ENG-Community Shield', 'TUR-Super Cup', 'FRA-Super Cup', 'GER-Super Cup', 'EU-UEFA Super Cup', 'BEL-Super Cup',
                      'ESP-Super Cup', 'NED-Super Cup', 'ITA-Supercoppa']:
            season = '2020/2021'

        # Create a FotMob object for the current league and season
        fotmob = sd.FotMob(leagues=league, seasons=season)

        # Read the schedule for the current league
        schedule = fotmob.read_schedule()

        # Add the schedule to the list
        all_schedules.append(schedule)
    except Exception as e:
        print(f"Error loading schedule for {league}: {e}")
        # If there is an error regarding a league, it will be skipped.

# Concatenate all schedules into a single DataFrame
if all_schedules:
    combined_schedule = pd.concat(all_schedules, ignore_index=True)
else:
    print("No schedules were successfully loaded.")

# Create empty columns for home and away rest times
combined_schedule['home_rest'] = pd.NaT
combined_schedule['away_rest'] = pd.NaT

# Get unique teams from both home and away columns
all_teams = pd.concat([combined_schedule['home_team'], combined_schedule['away_team']]).unique()

# Loop over each team
for team in all_teams:
    # Filter games for the team (whether home or away) and sort by date
    team_games = combined_schedule[
        (combined_schedule['home_team'] == team) | (combined_schedule['away_team'] == team)
    ].sort_values('date')
    
    # Calculate the time differences between consecutive games.
    # The first game for the team will naturally have NaT as there is no prior game.
    rest_times = team_games['date'].diff()
    
    # Iterate over the team's games and assign the computed rest times appropriately.
    for idx, game in team_games.iterrows():
        # If the team is playing at home in this game, update the home_rest column
        if game['home_team'] == team:
            combined_schedule.loc[idx, 'home_rest'] = rest_times.loc[idx]
        # If the team is playing away in this game, update the away_rest column
        if game['away_team'] == team:
            combined_schedule.loc[idx, 'away_rest'] = rest_times.loc[idx]

# Optionally, sort the entire DataFrame by date if needed
combined_schedule = combined_schedule.sort_values('date')

# Extract only the month
shots['month'] = shots['date'].dt.strftime('%m')

# Convert the month string to integer
shots['month_int'] = shots['month'].astype(int)

# Adjust month numbers: for months before August (1-7), add 12
shots['adjusted_month'] = shots['month_int'].apply(lambda m: m if m >= 8 else m + 12)

# Define bin edges for the adjusted month values:
#   - start of the season: August, September, October  (adjusted: 8, 9, 10)
#   - holiday season: November, December  (November=11, December=12, January becomes 13)
#   - start of the year: January, February, March  (February becomes 14, March=15, April=16)
#   - end of the season: April, May, June, July  (May becomes 17, June=18, July=19)
bin_edges = [7.5, 10.5, 12.5, 15.5, 19.5]
bin_labels = ["start of the season", "holiday season", "start of the year", "end of the season"]

# Categorize into period_category based on adjusted month
shots["period_category"] = pd.cut(shots["adjusted_month"], bins=bin_edges, labels=bin_labels, right=True)

# Optional: drop the temporary 'month_int' column if not needed
shots.drop(columns=["month_int"], inplace=True)

# Group by month to calculate average and median rest days
team_monthly_rest = shots.groupby('period_category').agg(
    avg_rest=('team_rest', lambda x: round(x.mean(), 2)),
    median_rest=('opponent_rest', lambda x: round(x.median(), 2)),
    num_matches=('game_id', 'nunique'),
    goals=("outcome", "sum"),
    shots=("type_name", "count"),
    xG=("xG", "sum"),
    w_xG=("weighted_xG", "sum"),
    G_xG=("G-xG", "sum"),
    G_w_xG=("G-w_xG", "sum"),
).reset_index()
